cmake_minimum_required(VERSION 3.28.0)
project(cqy VERSION 0.1.0 LANGUAGES C CXX)
set(CMAKE_CXX_STANDARD 20)  
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

add_executable(ping cqy.h ping.cpp)
target_include_directories(ping PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/3rd/yalantinglibs/include)
target_include_directories(ping PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/3rd/yalantinglibs/include/ylt/thirdparty)
target_include_directories(ping PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/3rd/yalantinglibs/include/ylt/standalone)

add_executable(pong cqy.h pong.cpp)
target_include_directories(pong PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/3rd/yalantinglibs/include)
target_include_directories(pong PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/3rd/yalantinglibs/include/ylt/thirdparty)
target_include_directories(pong PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/3rd/yalantinglibs/include/ylt/standalone)

if(CMAKE_CXX_COMPILER_ID MATCHES "GNU")
    message(STATUS "Using gcc compiler.")
    add_compile_options(-fsanitize=address -g)
    target_link_libraries(ping PRIVATE asan)
    target_link_libraries(pong PRIVATE asan)
else()
    add_compile_options("/wd4819")
    add_compile_options("/wd4267")
endif()
